class Solution {
    public int findMaxConsecutiveOnes(int[] nums) {
        int cnt = 0;
        int maxCnt = 0;
        
        for(int ele : nums){
            if(ele == 0){
                cnt = 0;
            }else{
                cnt++;
                maxCnt = Math.max(cnt,maxCnt);
            }
        }
        return maxCnt;
    }
}
